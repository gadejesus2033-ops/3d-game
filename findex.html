window.addEventListener('keydown', e => { if(e.repeat) return; const dir = new THREE.Vector3(); camera.getWorldDirection(dir); // flatten (ignore looking up/down) dir.y = 0; dir.normalize(); const right = new THREE.Vector3(); right.crossVectors(dir, new THREE.Vector3(0,1,0)).normalize(); let move = {x:0, z:0}; if(e.code === 'KeyW'){ move.x = Math.round(dir.x); move.z = Math.round(dir.z); } if(e.code === 'KeyS'){ move.x = -Math.round(dir.x); move.z = -Math.round(dir.z); } if(e.code === 'KeyA'){ move.x = -Math.round(right.x); move.z = -Math.round(right.z); } if(e.code === 'KeyD'){ move.x = Math.round(right.x); move.z = Math.round(right.z); } // prevent tiny values turning into 0 movement if(move.x !== 0 || move.z !== 0){ moveQueue.push(move); } });
