window.addEventListener('keydown', e => { if(e.repeat) return; const angle = camera.rotation.y; let dx = 0; let dz = 0; if(e.code === 'KeyW'){ dx = -Math.sin(angle); dz = -Math.cos(angle); } if(e.code === 'KeyS'){ dx = Math.sin(angle); dz = Math.cos(angle); } if(e.code === 'KeyA'){ dx = -Math.cos(angle); dz = Math.sin(angle); } if(e.code === 'KeyD'){ dx = Math.cos(angle); dz = -Math.sin(angle); } // snap to grid movement dx = Math.round(dx); dz = Math.round(dz); // 🔥 VERY IMPORTANT: fallback so it never becomes 0,0 if(dx === 0 && dz === 0){ if(e.code === 'KeyW') dz = -1; if(e.code === 'KeyS') dz = 1; if(e.code === 'KeyA') dx = -1; if(e.code === 'KeyD') dx = 1; } moveQueue.push({x:dx, z:dz}); // jump if(e.code === 'Space' && velocityY === 0){ velocityY = 0.25; } });
